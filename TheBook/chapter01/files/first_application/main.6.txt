inline void beforeApplication() {
    {
        /*初始化随机种子*/
        ::srand(static_cast<unsigned>(::time(nullptr)));
    }
    {
        /*高分屏支持*/
        QCoreApplication::setAttribute(Qt::AA_EnableHighDpiScaling);
    }
    {
        /*设置默认opengl环境*/
        QSurfaceFormat::setDefaultFormat(::getDefaultQSurfaceFormat());
    }
}
